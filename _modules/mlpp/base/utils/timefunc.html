from abc import ABCMeta, abstractmethod, abstractproperty

try:
    import matplotlib
    import matplotlib.pyplot as plt
    matplotlib_imported = True
except ImportError:
    matplotlib_imported = False
    

class _Plottable(metaclass=ABCMeta):
    """
    This abstract class can be inherited by any class which would like to be have a plot() function

    You must only override _plot function
    """

    def plot(self, ax=None, labels='default', **kwargs):
        """
        Plot this object

        :param ax: if filled, the object will be plot directly on it, otherwise a new ax will be created

        :param labels: labels you want to see attached to you figure, sometimes default labels exists

        :return: ax if you want to modify it afterwards
        """

        if not matplotlib_imported:
            raise(ImportError("You must have matplotlib.pyplot available"))

        ax, labels = self._pre_plot(ax, labels)
        self._plot(ax, labels, **kwargs)
        return self._finish_plot(ax, labels)

    @abstractmethod
    def _plot(self, ax, labels, **kwargs):
        pass

    def _pre_plot(self, ax, labels):
        if ax is None:
            ax = plt.subplot(111)
        self._no_labels = labels is None
        return ax, labels

    def _finish_plot(self, ax, labels):
        if not self._no_labels:
            ax.legend()
        return ax

    @staticmethod
    def _extended_discrete_xaxis(x_axis, num_points=100, eps=0.10):
        """Takes `x_axis` and returns an uniformly sampled array of values from its minimum to its maximum with few
            extra points

        :param x_axis:

        :param num_points: (default 100) the number of points in the sample

        :param eps: (default 0.05) number of extra points in percentage

        :return: the sample
        """
        min_value = np.min(x_axis)
        max_value = np.max(x_axis)
        distance = max_value - min_value
        return np.linspace(min_value - eps * distance, max_value + eps * distance, num=num_points)

    @staticmethod
    def color_map(value):
        import matplotlib.cm as cmx
        import matplotlib.colors as colors

        cm = plt.get_cmap('hsv')
        c_norm = colors.Normalize(vmin=0, vmax=1)
        scalar_map = cmx.ScalarMappable(norm=c_norm, cmap=cm)

        return scalar_map.to_rgba(value)



    [docs]class TimeFunction(_Plottable, _TimeFunction):
    """
    A function depending on time.

    It is causal as its value is zero for all :math:`t < 0`.

    :param values:
        Either a number if the TimeFunction is constant or a list of two numpy arrays ``[T, Y]`` where
        ``Y`` are the values taken by the TimeFunction at times ``T``

    :param border_type:
        Handle the values returned after the after the last given ``t`` (non constant case)

        * ``Border0`` : value will be :math:`0`
        * ``BorderConstant`` : value will be given by ``border_value``
        * ``BorderConstant`` : value will equal to the last known value

    :param inter_mode:
        Handle the way we extrapolate between two known values (non constant case)

        * ``InterLinear`` : value will be linearly interpolated following the formula :math:`f(x) = \\frac{y_{t+1} - y_{t}}{x_{t+1} - x_{t}}`
        * ``InterConstLeft`` : value will be equal to the next known point
        * ``InterConstRight`` : value will equal to the previous known point

    :param dt:
        The value used for the sub-sampling. If left to 0, it will be assigned automatically to a fifth of the smallest
        distance between two points

    :param border_value:
        See ``border_type``, ``BorderConstant`` case

    .. note:: TimeFunction are made to be every efficient when call if to get a specific value (:math:`\mathcal{O}(1)`),
        however this leads us to have it taking a lot of space in memory.
    """

    def __init__(self, values,
                 border_type: int=_TimeFunction.Border0,
                 inter_mode: int=_TimeFunction.InterLinear,
                 dt: float=0, border_value: float=0):

        if isinstance(values, (int, float)):
            super(TimeFunction, self).__init__(values)
            self.is_constant = True
        else:
            t_array = values[0]
            y_array = values[1]
            super(TimeFunction, self).__init__(t_array, y_array, border_type, inter_mode, dt, border_value)
            self.original_y = y_array
            self.original_t = t_array
            self.is_constant = False

    @property
    def defaults_labels(self):
        if self.is_constant:
            return ['value = ']

        else:
            interpolation_to_legend = {_TimeFunction.InterLinear: 'Linear',
                                       _TimeFunction.InterConstLeft: 'Constant on left',
                                       _TimeFunction.InterConstRight: 'Constant on right'}

            border_to_legend = {_TimeFunction.Border0: 'border zero',
                                _TimeFunction.BorderConstant: 'border constant at %.3g' % self.border_value,
                                _TimeFunction.BorderContinue: 'border continue'}

            return ['original points', '%s and %s' % (interpolation_to_legend[self.inter_mode],
                                                      border_to_legend[self.border_type])]

        """Plot the TimeFunction

        :param ax: if filled, the object will be plot directly on it, otherwise a new ax will be created

        :param labels: labels you want to see attached to you figure, you can leave it to default for automatic fill
            or None if you do not want any.

        :param num_points: The number of points (evenly distributed) you want your function to be plotted with.

        :return: ax if you want to modify it afterwards
        """
        _Plottable.plot(self, ax=ax, labels=labels, num_points=num_points)
    

    def _plot(self, ax, labels, num_points=100):
        if labels.__class__ != list or len(labels) != 2:
            labels = self.defaults_labels

        if self.is_constant:
            t_values = np.arange(10).astype('float')
            ax.plot(t_values, self.value(t_values), label=labels[0])
        else:
            t_values = self._extended_discrete_xaxis(self.original_t, num_points=num_points)
            ax.plot(self.original_t, self.original_y, ls='', marker='o', label=labels[0])
            ax.plot(t_values, self.value(t_values), label=labels[1])
