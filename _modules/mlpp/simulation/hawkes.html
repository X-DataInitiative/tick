[docs]class SimuHawkes(SimuPointProcess, _Plottable):
    """Class for simulation of Hawkes process
    
    They are defined by the intensity:
    
    .. math::
        \forall i \in [1 \dots D], \quad
        \lambda_i(t) = \mu_i + \sum_{j=1}^D \int \phi_{ij}(t - s) dN_j(s)

    where
    
      - :math:`D` is the number of nodes
      - :math:`\mu_i` are the baseline intensities
      - :math:`\phi_{ij}` are the kernels
      - :math:`dN_j` are the processes differentiates

    Parameters
    ----------
    kernels : `np.ndarray`, shape=(n_nodes, n_nodes)
        A 2-dimensional arrays of kernels, also noted :math:`\phi_{ij}`

    baselines : `np.ndarray`, shape=(n_nodes, )
        The baseline of all intensities, also noted :math:`\mu`

    n_nodes : `int`
        The number of nodes of the Hawkes process. If kernels and baselines
        are None this will create a Hawkes process with n_nodes nodes of zero
        kernels and with zero baselines

    end_time : `float`, default=None
        Time until which this point process will be simulated

    max_jumps : `int`, default=None
        Simulation will stop if this number of jumps in reached

    seed : `int`, default = None
        The seed of the random sampling. If it is None then a random seed
        (different at each run) will be chosen.

    force_simulation : `bool`, default = False
        If force is not set to True, simulation won't be run if the matrix of
        the L1 norm of kernels has a spectral radius greater or equal to 1 as
        it would be unstable

    Attributes
    ----------
    process : `list` of `np.ndarray`, size=n_nodes
        A list of n_nodes processes, each list being the times of all the
        jumps for this coordinate

    simulation_time : `float`
        Time until which this point process has been simulated

    n_total_jumps : `int`
        Total number of jumps simulated

    tracked_intensity : `list[np.ndarray]`, size=n_nodes
        A record of the intensity with which this point process has been
        simulated.
        Note: you must call track_intensity before simulation to record it

    intensity_tracked_times : `np.ndarray`
        The times at which intensity has been recorded.
        Note: you must call track_intensity before simulation to record it

    intensity_track_step : `float`
        Step with which the intensity has been recorded
    """

    _attrinfos = {
        "kernels": {"writable": False},
        "_kernel_0": {"writable": False},
    }

    def __init__(self, kernels=None, baselines=None, n_nodes=None,
                 end_time=None, max_jumps=None, seed=None, verbose=True,
                 force_simulation=False):
        SimuPointProcess.__init__(self, end_time=end_time, max_jumps=max_jumps,
                                  seed=seed, verbose=verbose)

        self.force_simulation = force_simulation
        # We keep a reference on this kernel to avoid copies
        self._kernel_0 = HawkesKernel0()

        if isinstance(kernels, list):
            kernels = np.array(kernels)

        if isinstance(baselines, list):
            baselines = np.array(baselines)

        if baselines is not None and baselines.dtype != float:
            baselines = baselines.astype(float)

        self.check_parameters_coherence(kernels, baselines, n_nodes)

        # Init _pp so we hae access to self.n_nodes
        if n_nodes is None:
            if baselines is not None:
                n_nodes = baselines.shape[0]
            else:
                n_nodes = kernels.shape[0]

        if n_nodes <= 0:
            raise ValueError("n_nodes must be positive but equals %i" % n_nodes)
        self._pp = _Hawkes(n_nodes, self._pp_seed)

        if kernels is not None:
            if kernels.shape != (self.n_nodes, self.n_nodes):
                raise ValueError("kernels shape should be %s instead of %s" %
                                 ((self.n_nodes, self.n_nodes),
                                  kernels.shape))
            self.kernels = kernels
            self._init_kernels()
        else:
            self._init_zero_kernels()

        if baselines is not None:
            if baselines.shape != (self.n_nodes,):
                raise ValueError("baselines shape should be %s instead of %s" %
                                 ((self.n_nodes,), self.baselines.shape))
            self.baselines = baselines
            self._init_baselines()

        else:
            self._init_zero_baselines()

    def _init_zero_kernels(self):
        self.kernels = np.empty((self.n_nodes, self.n_nodes), dtype=object)
        for i, j in product(range(self.n_nodes), range(self.n_nodes)):
            self.kernels[i, j] = self._kernel_0

    def _init_zero_baselines(self):
        self.baselines = np.empty((self.n_nodes,), dtype=object)
        for i in range(self.n_nodes):
            self.set_baseline(i, 0)

    def _init_baselines(self):
        for i in range(self.n_nodes):
            self.set_baseline(i, self.baselines[i])

    def _init_kernels(self):
        for i, j in product(range(self.n_nodes), range(self.n_nodes)):
            if self.kernels[i, j] == 0:
                kernel_ij = self._kernel_0
            else:
                kernel_ij = self.kernels[i, j]
            self.set_kernel(i, j, kernel_ij)

    def check_parameters_coherence(self, kernels, baselines,  n_nodes):
        set_kernels = kernels is not None
        set_baselines = baselines is not None
        set_n_nodes = n_nodes is not None

        if set_n_nodes and (set_kernels or set_baselines):
            raise ValueError("n_nodes will be automatically calculated if "
                             "baselines or kernels is set")

        if not set_n_nodes and not set_kernels and not set_baselines:
            raise ValueError("n_nodes must be given if neither kernels, "
                             "nor baselines are given")

        if set_kernels and set_baselines and len(kernels) != len(baselines):
            raise ValueError("kernels and baselines have different length. "
                             "kernels has length %i, whereas baselines has "
                             "length %i." % (len(kernels), len(baselines)))

    def set_kernel(self, i, j, kernel):
        if isinstance(kernel, np.number) and kernel == 0:
            self.kernels[i, j] = self._kernel_0
            self._pp.set_kernel(i, j, self._kernel_0._kernel)
        else:
            self.kernels[i, j] = kernel
            self._pp.set_kernel(i, j, kernel._kernel)

    def set_baseline(self, i, baseline):
        self.baselines[i] = baseline
        self._pp.set_mu(i, baseline)

    def _simulate(self):
        """Launch simulation of the Hawkes process by thinning
        """
        if np.linalg.norm(self.baselines) == 0:
            warnings.warn("Baselines have not been set, hence this hawkes "
                          "process won't tick")

        if not self.force_simulation and self.spectral_radius() >= 1 \
                and self.max_jumps is None:
            raise ValueError("Simulation not launched as this Hawkes process "
                             "is not stable (spectral radius of %.2g). You "
                             "can use force_simulation parameter if you "
                             "really want to simulate it"
                             % self.spectral_radius())

        SimuPointProcess._simulate(self)


[docs]    def spectral_radius(self):
        """Compute the spectral radius of the matrix of l1 norm of Hawkes
        kernels.

        Notes
        -----
        If the spectral radius is greater that 1, the hawkes process is not
        stable
        """

        get_norm = np.vectorize(lambda kernel: kernel.get_norm())
        norms = get_norm(self.kernels)

        # It might happens that eig returns a complex number but with a
        # negligible complex part, in this case we keep only the real part
        spectral_radius = max(eig(norms)[0])
        spectral_radius = np.real_if_close(spectral_radius)
        return spectral_radius



[docs]    def mean_intensity(self):
        """Compute the mean intensity vector
        """
        get_norm = np.vectorize(lambda kernel: kernel.get_norm())
        norms = get_norm(self.kernels)
        return inv(np.eye(self.n_nodes) - norms).dot(self.baselines)

    def defaults_labels(self):
        return ["jumps", "intensity"]

             t_min: float = None, t_max: float = None,
             max_points: int = np.inf):
        """Quick plot of the Hawkes process

        Parameters
        ----------
        ax :
            if filled, the object will be plot directly on it, otherwise a new
            ax will be created

        labels : `list[str]`
            labels you want to see attached to you figure, you can leave it to
            default for automatic fill or None if you do not want any.

        dim : `int`
            The selected node

        Returns
        -------
            ax in case you want to modify it afterwards
        """
        _Plottable.plot(self, ax, labels, style=style, dim=dim, t_min=t_min,
                        t_max=t_max, max_points=max_points)


    def _plot(self, ax, labels, style='time', dim=None, t_min: float = None,
              t_max: float = None, max_points: int = np.inf):

        if style == 'time' or style == 'over_time':
            if self._pp.itr_on():
                _intensity_ticks_plot(
                    self, ax, labels, title='Hawkes process', dim=dim,
                    delay=self.intensity_track_step,
                    considered_dims=set(range(self.n_nodes)),
                    t_min=t_min, t_max=t_max, max_points=max_points)
            else:
                _ticks_plot(self, ax, labels, title='Hawkes process',
                            considered_dims=set(range(self.n_nodes)),
                            t_min=t_min, t_max=t_max, max_points=max_points)
        elif style == 'ticks':
            _ticks_plot(self, ax, labels, title='Hawkes process',
                        considered_dims=set(range(self.n_nodes)),
                        t_min=t_min, t_max=t_max, max_points=max_points)

        else:
            raise ValueError('Unknown plot style')

    def plot_kernels(self, support=-1):

        import matplotlib.pyplot as plt

        # First we need to get the max support
        if support == -1:
            get_plot_supports = np.vectorize(
                lambda kernel: kernel.get_plot_support())
            plot_supports = get_plot_supports(self.kernels)
            support = plot_supports.max()

        x = np.arange(0, support, 1.2 * support / 3000.)
        fig, ax_list_list = plt.subplots(
            self.n_nodes, self.n_nodes, sharex=True, sharey=True)

        if self.n_nodes == 1:
            ax_list_list = np.array([[ax_list_list]])

        for i, ax_list in enumerate(ax_list_list):
            for j, ax in enumerate(ax_list):
                y = self.kernels[i, j].get_values(x)
                ax.plot(x, y, label=self.kernels[i, j].__strtex__())
                x_label = r"$t$"
                y_label = r"$\phi^{%g,%g}(t)$" % (i, j)
                ax.set(xlabel=x_label, ylabel=y_label)
                ax.labelsize = 35
                legend = ax.legend()
                for label in legend.get_texts():
                    label.set_fontsize(12)
